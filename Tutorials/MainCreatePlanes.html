Sort node planes

The objective of this function is to take as input the original list of nodes from the NASA CRM model, sort the ones belonging to the rib planes and reindex them.
Read the input text files
clearvars
close all
clc
nodes = ImportNodes('nodes.txt');
NodesSection = ImportNodesSection('nodesSection.txt');
massNodes = ImportMass('mass.txt');
save nodes.mat nodes
save massNodes.mat massNodes
save NodesSection.mat NodesSection
N = size(nodes,1);
Remove the concentrated mass nodes
for i = 1:length(massNodes)
found = true;
p = 1;
while found
if nodes(p,1) == massNodes(i)
found = false;
NodesM(i,:) = nodes(p,:);
nodes(p,:) = [];
end
p = p + 1;
end
end
Find the first 56 ribs (i up to 59 is required due to catch them all) and store them in Planes
nodelist = nodes;
k = 1;
for i = 1:59
[inP,outP,planevar] = findPlane(nodelist,NodesSection(:,2:3),2*10^-2,1);
nodelist = outP;
if k ~= 1
a = size(Planes(k-1).coord,1);
end
if size(inP,1) < 140 && a < 140
Planes(k-1).coord(a+1:a+size(inP,1),:) = inP;
else
Planes(k).coord = inP;
Planes(k).plane = planevar;
k = k + 1;
end
end
Remove nodes by their x coordinate in order to speed up the search for the last two ribs
r = 1;
for j = 1 : size(outP,1)
if outP(j,3) < 28.641
LNind(r) = j;
r = r + 1;
end
end
LeftNodes = outP(LNind,:);
outP(LNind,:) = [];
nodelist = outP;
Store the last two ribs
for i = 60:61
[inP,outP] = findPlane(nodelist,NodesSection(:,2:3),1*10^-2,3);
nodelist = outP;
Planes(k).coord = inP;
Planes(k).plane = planevar;
k = k + 1;
end
Save the final list
LeftNodes = [LeftNodes; outP];
save LeftNodes.mat LeftNodes
Define a new structure for the planes
for i = 1:58
nP = projectPlane(Planes(i));
nPS = nodeSort(nP);
FP(i).localRef = nPS(:,1);
for j = 1:size(nPS,1)
FP(i).coord(j,:) = findRealCoord(nPS(j,2),nodes);
end
end
Sort the planes from root to tip
for i = 1:58
yL(i) = FP(i).coord(1,2);
end
[~,idx] = sort(yL); % sort just the first column
FP = FP(idx);
save FP.mat FP
Plot the ribs
figure()
for i = 1:58
scatter3(FP(i).coord(:,1),FP(i).coord(:,2),FP(i).coord(:,3))
hold on
end
grid minor
xlabel('X')
ylabel('Y')
zlabel('Z')
